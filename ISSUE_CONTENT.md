```yaml
# PR 流程：只跑单元测试（5 分钟）
on:
  pull_request:
    jobs:
      - backend-unit-test
      - frontend-unit-test

# main/develop 分支：跑全量测试（15 分钟）
on:
  push:
    branches: [main, develop]
    jobs:
      - backend-full-test
      - frontend-full-test
      - security-scan
      - deploy-staging
```
