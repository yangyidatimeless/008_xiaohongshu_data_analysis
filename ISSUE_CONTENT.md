```typescript
class WebSocketClient {
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;
  private reconnectDelays = [1000, 2000, 4000, 8000, 16000, 32000];

  connect() {
    this.ws = new WebSocket('ws://test-server:8000/ws/notifications');
    
    this.ws.onclose = () => {
      if (this.reconnectAttempts < this.maxReconnectAttempts) {
        const delay = this.reconnectDelays[this.reconnectAttempts];
        console.log(`WebSocket 断开，${delay}ms 后重连...`);
        setTimeout(() => {
          this.reconnectAttempts++;
          this.connect();
        }, delay);
      } else {
        showToast('连接服务器失败，请刷新页面重试', 'error');
      }
    };
    
    this.ws.onopen = () => {
      this.reconnectAttempts = 0; // 重置重连计数
    };
  }
}

// src/utils/request.ts - Axios 拦截器
import axios from 'axios';
import { showToast } from '@/components/Toast';

const request = axios.create({
  baseURL: '/api/v1',
  timeout: 5000,
});

// 响应拦截器 - 统一错误处理
request.interceptors.response.use(
  (response) => response.data,
  (error) => {
    const { response } = error;
    
    if (response) {
      // 后端返回的错误
      const { code, message } = response.data;
      
      switch (response.status) {
        case 400:
          showToast(`请求参数错误：${message}`, 'warning');
          break;
        case 401:
          showToast('未授权，请登录', 'error');
          // 跳转到登录页
          break;
        case 403:
          showToast('权限不足', 'error');
          break;
        case 404:
          showToast('资源不存在', 'error');
          break;
        case 500:
          showToast(`服务器错误：${message}`, 'error');
          break;
        default:
          showToast(message || '请求失败', 'error');
      }
    } else {
      // 网络错误
      showToast('网络连接失败，请检查网络', 'error');
    }
    
    return Promise.reject(error);
  }
);

export default request;
```
